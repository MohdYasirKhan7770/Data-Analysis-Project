Quick Start

Get Started In Seconds

Simple, intuitive API that makes data analysis effortless.

analysis.py 
# Basic Data Analysis with Prismlytics
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load and explore dataset
df = pd.read_csv("sales_data.csv")
print(df.describe())
print(df.info())

# Visualize revenue trends
plt.figure(figsize=(12, 6))
sns.lineplot(data=df, x="date", y="revenue")
plt.title("Revenue Over Time")
plt.savefig("output/revenue_chart.png")
plt.show()
